function exportPDF() { const logo = new Image(); logo.src = "Arrington_Financial_logo-removebg.png"; logo.onload = function() { html2canvas(document.querySelector(".container")).then(canvas => { const { jsPDF } = window.jspdf; const pdf = new jsPDF(); const imgData = canvas.toDataURL("image/png"); const imgWidth = 180; const imgHeight = (canvas.height * imgWidth) / canvas.width; // Add logo pdf.addImage(logo, "PNG", 10, 10, 50, 20); // Add content pdf.addImage(imgData, "PNG", 10, 40, imgWidth, imgHeight); // Add second page for disclosure pdf.addPage(); pdf.setFontSize(12); pdf.text("Disclosures:", 10, 20); pdf.setFontSize(10); const disclosureText = ` The analysis provided by this tool is based solely on the information provided by you... (Truncated for brevity) `; pdf.text(disclosureText, 10, 30, { maxWidth: 180 }); pdf.save("Retirement_Calculator.pdf"); }); }; logo.onerror = function() { console.error("Failed to load logo image."); }; }
